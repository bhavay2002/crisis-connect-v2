// POST /outcomes
export async function recordOutcome(req, res) {
  const { decisionId, outcome, responseTime } = req.body;

  await db.outcomes.create({
    decisionId,
    outcome,
    responseTime,
    recordedAt: new Date(),
  });

  // emit for analytics + training
  eventBus.emit("outcome.recorded", { decisionId, outcome, responseTime });

  res.json({ ok: true });
}

# 🧪 7) ENFORCED TESTING + CI (NON-NEGOTIABLE)

## 🔷 GitHub Actions CI

```yaml id="ci1"
name: CI

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v3

      - name: Install
        run: npm install

      - name: Run Tests
        run: npm run test -- --coverage

      - name: Enforce Coverage
        run: |
          COVERAGE=$(jq '.total.lines.pct' coverage/coverage-summary.json)
          if (( $(echo "$COVERAGE < 80" | bc -l) )); then
            echo "Coverage below 80%"
            exit 1
          fi
```
