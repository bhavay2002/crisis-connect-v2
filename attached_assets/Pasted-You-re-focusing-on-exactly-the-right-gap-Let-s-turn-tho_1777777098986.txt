## 🔷 Frontend Implementation (React + Zustand)

### Store

```ts
// store/decisionStore.ts
import { create } from "zustand";

export const useDecisionStore = create((set) => ({
  activeDecision: null,
  setDecision: (decision) => set({ activeDecision: decision }),
}));
```

---

### Action Panel Component

```tsx
// features/crisis/ActionPanel.tsx
import { motion } from "framer-motion";
import { useDecisionStore } from "@/store/decisionStore";

export default function ActionPanel() {
  const decision = useDecisionStore((s) => s.activeDecision);

  if (!decision) return null;

  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      className="fixed bottom-6 right-6 w-96 bg-red-600 text-white rounded-2xl shadow-xl p-5"
    >
      <h3 className="text-lg font-bold">🚨 Critical Incident</h3>
      <p className="text-sm opacity-90">
        {decision.location} • Severity {decision.severity}
      </p>

      <div className="mt-4 space-y-2">
        {decision.recommendedActions.map((action) => (
          <button
            key={action.id}
            className={`w-full py-2 rounded-xl ${
              action.type === "primary"
                ? "bg-white text-red-600"
                : "bg-red-500"
            }`}
          >
            {action.label}
          </button>
        ))}
      </div>
    </motion.div>
  );
}
```
