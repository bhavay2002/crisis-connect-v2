interface Decision {

  type: "DISPATCH" | "ESCALATE" | "BROADCAST" | "PREDEPLOY";

  severity: "LOW" | "MEDIUM" | "HIGH" | "CRITICAL";

    repetition: number;

  };

  autoExecutable: boolean;
  status: "PENDING" | "APPROVED" | "EXECUTED" | "REJECTED";

  createdAt: Date;
}

if (fusedScore > 0.8) {
